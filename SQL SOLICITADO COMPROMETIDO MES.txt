SELECT 
to_char(OICD.AUTORIZACIONES_ADD.FE_RECEPCION,'YYYYMM') AS 'MES', 
(SUM(OICD.DETALLE_EGRESO.MO_EGRESO)/1000000) AS 'SOLICITADO'

FROM 
"OICD".AUTORIZACIONES AUTORIZACIONES, 
"OICD".AUTORIZACIONES_ADD AUTORIZACIONES_ADD, 
"OICD".DETALLE_EGRESO DETALLE_EGRESO

WHERE 
(OICD.AUTORIZACIONES_ADD.NU_AUTORIZ_ADD=OICD.AUTORIZACIONES.NU_AUTORIZ_ADD) AND 
(OICD.AUTORIZACIONES.NU_AUTORIZACION=OICD.DETALLE_EGRESO.NU_AUTORIZACION) AND 
(OICD.AUTORIZACIONES.FE_RECEPCION<=SYSDATE-3) AND (OICD.DETALLE_EGRESO.FE_REGISTRO<=SYSDATE-3)

GROUP BY 
to_char(OICD.AUTORIZACIONES_ADD.FE_RECEPCION,'YYYYMM')
